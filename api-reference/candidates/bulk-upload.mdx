## Overview

Bulk upload candidate resumes in PDF, DOC, or DOCX format. The system automatically parses resumes, extracts candidate information, and scores them against job requirements.

## Endpoint

```
POST /bulk-upload-resumes
Content-Type: multipart/form-data
```

## Request Parameters

<ParamField body="files" type="file[]" required>
  Array of resume files (PDF/DOC/DOCX format)
</ParamField>

<ParamField body="jobId" type="string" required>
  Job ID to associate candidates with
</ParamField>

<ParamField body="recruiterId" type="string" required>
  Recruiter's unique identifier
</ParamField>

<ParamField body="organizationId" type="string" required>
  Organization's unique identifier
</ParamField>

<ParamField body="userType" type="string" required>
  User role (Admin/Recruiter)
</ParamField>

<ParamField body="resumeScoreBreakdown" type="string" required>
  JSON string containing scoring criteria for resume evaluation
</ParamField>

## Request Example

<CodeGroup>

```bash cURL
curl -X POST https://api.exterview.ai/ats/candidates/bulk-upload \
  -H "x-api-key: YOUR_API_KEY" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -F "files=@resume1.pdf" \
  -F "files=@resume2.pdf" \
  -F "files=@resume3.pdf" \
  -F "jobId=688767b8477f8d30dc51b6d5" \
  -F "recruiterId=68a5b11168f3d6ccd1a" \
  -F "organizationId=68a595060754e2425" \
  -F "userType=Admin" \
  -F 'resumeScoreBreakdown={"experience": 40, "skills": 35, "education": 25}'
```

```javascript JavaScript
const formData = new FormData();

// Add multiple files
const fileInput = document.querySelector('#resume-files');
for (let file of fileInput.files) {
  formData.append('files', file);
}

formData.append('jobId', '688767b8477f8d30dc51b6d5');
formData.append('recruiterId', '68a5b11168f3d6ccd1a');
formData.append('organizationId', '68a595060754e2425');
formData.append('userType', 'Admin');
formData.append('resumeScoreBreakdown', JSON.stringify({
  experience: 40,
  skills: 35,
  education: 25
}));

const response = await fetch('https://api.exterview.ai/ats/candidates/bulk-upload', {
  method: 'POST',
  headers: {
    'x-api-key': 'YOUR_API_KEY',
    'Authorization': 'Bearer YOUR_TOKEN'
  },
  body: formData
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.exterview.ai/ats/candidates/bulk-upload"
headers = {
    "x-api-key": "YOUR_API_KEY",
    "Authorization": "Bearer YOUR_TOKEN"
}

files = [
    ('files', open('resume1.pdf', 'rb')),
    ('files', open('resume2.pdf', 'rb')),
    ('files', open('resume3.pdf', 'rb'))
]

data = {
    'jobId': '688767b8477f8d30dc51b6d5',
    'recruiterId': '68a5b11168f3d6ccd1a',
    'organizationId': '68a595060754e2425',
    'userType': 'Admin',
    'resumeScoreBreakdown': '{"experience": 40, "skills": 35, "education": 25}'
}

response = requests.post(url, headers=headers, files=files, data=data)
print(response.json())
```

## Response

<ResponseField name="status" type="integer">
  Status code: 1 for success, 0 for failure
</ResponseField>

<ResponseField name="message" type="string">
  Response message describing the result
</ResponseField>

<ResponseField name="created" type="integer">
  Number of candidates successfully created
</ResponseField>

<ResponseField name="duplicates" type="integer">
  Number of duplicate candidates skipped
</ResponseField>

<ResponseField name="skipped" type="integer">
  Number of resumes skipped due to errors
</ResponseField>

<ResponseField name="candidates" type="array">
  Array of created candidate objects
  
  <Expandable title="Candidate Object">
    <ResponseField name="candidate_unique_id" type="string">
      Unique identifier for the candidate
    </ResponseField>
    
    <ResponseField name="email" type="string">
      Candidate's email address
    </ResponseField>
    
    <ResponseField name="jobId" type="string">
      Associated job ID
    </ResponseField>
  </Expandable>
</ResponseField>
