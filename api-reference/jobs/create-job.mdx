## Overview

Create new job postings by uploading job definition files in CSV, XLSX, or JSON format. Optionally provide a prompt for additional parsing instructions.

## Endpoint

```
POST /addJobByFile
Content-Type: multipart/form-data
```

## Request Parameters

<ParamField body="recfiles" type="file" required>
  Job definition file in CSV, XLSX, or JSON format
</ParamField>

<ParamField body="recruiterId" type="string" required>
  Recruiter's unique identifier
</ParamField>

<ParamField body="organizationId" type="string" required>
  Organization identifier
</ParamField>

<ParamField body="prompt" type="string">
  Additional parsing instructions for the AI to interpret the job file
</ParamField>

## Request Example

<CodeGroup>

```bash cURL
curl -X POST https://api.exterview.ai/ats/jobs/upload \
  -H "x-api-key: YOUR_API_KEY" \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -F "recfiles=@job_posting.csv" \
  -F "recruiterId=68a5b11168f3d6ccd1a" \
  -F "organizationId=68a595060754e2425" \
  -F "prompt=Parse as software engineering roles"
```

```javascript JavaScript
const formData = new FormData();
formData.append('recfiles', fileInput.files[0]);
formData.append('recruiterId', '68a5b11168f3d6ccd1a');
formData.append('organizationId', '68a595060754e2425');
formData.append('prompt', 'Parse as software engineering roles');

const response = await fetch('https://api.exterview.ai/ats/jobs/upload', {
  method: 'POST',
  headers: {
    'x-api-key': 'YOUR_API_KEY',
    'Authorization': 'Bearer YOUR_TOKEN'
  },
  body: formData
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.exterview.ai/ats/jobs/upload"
headers = {
    "x-api-key": "YOUR_API_KEY",
    "Authorization": "Bearer YOUR_TOKEN"
}

files = {
    'recfiles': open('job_posting.csv', 'rb')
}

data = {
    'recruiterId': '68a5b11168f3d6ccd1a',
    'organizationId': '68a595060754e2425',
    'prompt': 'Parse as software engineering roles'
}

response = requests.post(url, headers=headers, files=files, data=data)
print(response.json())
```

## Response

<ResponseField name="status" type="integer">
  Status code: 1 for success, 0 for failure
</ResponseField>

<ResponseField name="message" type="string">
  Response message describing the result
</ResponseField>

<ResponseField name="jobs" type="array">
  Array of created job objects
  
  <Expandable title="Job Object Properties">
    <ResponseField name="job_unique_id" type="string">
      Unique identifier for the job
    </ResponseField>
    
    <ResponseField name="job_title" type="string">
      Title of the job position
    </ResponseField>
    
    <ResponseField name="job_code" type="string">
      Internal job reference code
    </ResponseField>
    
    <ResponseField name="department" type="string">
      Department or team for the position
    </ResponseField>
    
    <ResponseField name="location" type="string">
      Job location
    </ResponseField>
    
    <ResponseField name="employment_type" type="string">
      Type of employment (Full-time, Part-time, Contract, etc.)
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      ISO 8601 timestamp of job creation
    </ResponseField>
  </Expandable>
</ResponseField>
