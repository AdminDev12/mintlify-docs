## Overview

Get a ranked leaderboard of candidates for a job, sorted by their scores. Supports both AI-generated and manual scoring.

## Endpoint

```
POST /graphql
Content-Type: application/json
```

## GraphQL Query

```graphql
query getLeaderboard($input: InputTypeLeaderBoard!) {
  getLeaderboard(input: $input) {
    rows {
      name
      value
    }
  }
}
```

## Request Parameters

<ParamField body="jobId" type="string" required>
  Unique identifier of the job
</ParamField>

<ParamField body="organizationId" type="string" required>
  Organization identifier
</ParamField>

<ParamField body="leaderboardtype" type="string" required>
  Type of leaderboard: "ai" for AI-generated scores or "manual" for manual reviewer scores
</ParamField>

```javascript JavaScript
const query = `
  query getLeaderboard($input: InputTypeLeaderBoard!) {
    getLeaderboard(input: $input) {
      rows {
        name
        value
      }
    }
  }
`;

const variables = {
  input: {
    jobId: "68c42431b9f38f8c24fbc91f",
    leaderboardtype: "ai", // or "manual"
    organizationId: "68a5bac35f995060754e2425"

const response = await fetch('https://api.exterview.ai/graphql', {
  method: 'POST',
  headers: {
    'x-api-key': 'YOUR_API_KEY',
    'Authorization': 'Bearer YOUR_TOKEN',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({ query, variables })
});

const data = await response.json();
```

```python Python
import requests

url = "https://api.exterview.ai/graphql"
headers = {
    "x-api-key": "YOUR_API_KEY",
    "Authorization": "Bearer YOUR_TOKEN",
    "Content-Type": "application/json"
}

query = """
query getLeaderboard($input: InputTypeLeaderBoard!) {
  getLeaderboard(input: $input) {
    rows {
      name
      value
    }
  }
}
"""

variables = {
    "input": {
        "jobId": "68c42431b9f38f8c24fbc91f",
        "leaderboardtype": "ai",  # or "manual"
        "organizationId": "68a5bac35f995060754e2425"
    }
}

response = requests.post(url, headers=headers, json={"query": query, "variables": variables})
print(response.json())
```
